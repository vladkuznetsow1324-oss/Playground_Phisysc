const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // Растягиваем холст на всё окно canvas.width = window.innerWidth; canvas.height = window.innerHeight; // Настройки кубика let box = { x: 100, y: 100, size: 60, color: '#ff4757', isDragging: false }; function isMouseInBox(mx, my, b) { return mx > b.x && mx < b.x + b.size && my > b.y && my < b.y + b.size; } canvas.addEventListener('mousedown', (e) => { if (isMouseInBox(e.offsetX, e.offsetY, box)) { box.isDragging = true; } }); window.addEventListener('mousemove', (e) => { if (box.isDragging) { box.x = e.offsetX - box.size / 2; box.y = e.offsetY - box.size / 2; } }); window.addEventListener('mouseup', () => { box.isDragging = false; }); // Функция отрисовки function draw() { // Очищаем экран под размер окна ctx.clearRect(0, 0, canvas.width, canvas.height); // Рисуем кубик ctx.fillStyle = box.color; ctx.fillRect(box.x, box.y, box.size, box.size); requestAnimationFrame(draw); } // Запускаем draw();
